%matplotlib inline
import matplotlib.pyplot as plt
import math
import numpy

def normale_densite(x,mu):
    c=1.0/math.sqrt(2*math.pi)
    return c*math.exp(-0.5*(x-mu)**2)

fig, axes = plt.subplots(nrows=2, ncols=3,figsize=(10, 8))
#plt.tight_layout() # Or equivalently,  "fig.tight_layout()"

mu=[[0,-1,1],[-3,-2,7]]

for i in range(len(axes)):
    for j in range(len(axes[i])):
        # Cacher le cadre droit et haut
        axes[i][j].spines['right'].set_visible(False)
        axes[i][j].spines['top'].set_visible(False)
        # cacher les ticks
        axes[i][j].xaxis.set_ticks_position('bottom')
        axes[i][j].yaxis.set_ticks_position('left')
        
        XX=numpy.linspace(mu[i][j]-3.5,mu[i][j]+3.5,200)
        YY=[normale_densite(x,mu[i][j]) for x in XX]
        axes[i][j].plot(XX,YY)
        
        axes[i][j].set_title("$\\mu=$%s"%(mu[i][j]))
        
        axes[i][j].set_xlabel("$x$")
        axes[i][j].set_ylabel("$\\varphi(x-\\mu)$")
        

plt.subplots_adjust(top=0.92, bottom=0.19, left=0.1, right=0.98, hspace=0.35,wspace=0.65)

plt.show()
